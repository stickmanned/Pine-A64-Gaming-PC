# Forces the display to negotiate at 1080p during boot, before the OS loads.
# This keeps the shared memory pool (CMA) small enough to leave room for the
# hardware video decoder (CedarX VPU) to initialize instead of falling back
# to software decoding. See Hour 5 and Hour 6 of the devlog.
video=HDMI-A-1:1920x1080@60
